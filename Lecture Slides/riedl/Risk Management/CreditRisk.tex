\section{Modelling Challenge}
\frame{\frametitle{What is Credit Risk?}
Distinguish between \textcolor{blue}{individual risk elements}

\begin{itemize}
\item<1-> {\bf Default Probability / Credit Migration} The probability that the obligor
or counterparty will default on its contractual obligations to
repay its debt or the probability that its credit quality changes. Important is the time of default $\tau$.
\item<2-> {\bf Recovery Rates $\delta$ (R).} The extent to which the face value of
an obligation can be recovered once the obligor has defaulted.
\item<3-> {\bf Exposure at Default.} The amount the creditor is owed at the time of default.
\end{itemize}
}
\frame{\frametitle{What is Credit Risk?}

and \textcolor{blue}{portfolio risk elements}

\begin{itemize}
\item<1-> {\bf Default and Credit Quality Correlation.} The degree to
which the default or credit quality of one obligor is related to
the default or credit quality of another.
\item<2-> {\bf Risk Contribution and Credit Concentration.} The extent
to which an individual instrument or the presence of an obligor in
the portfolio contributes to the totality of risk in the overall
portfolio.
\end{itemize}
}

\section{Objects and Notation}
\subsection{Definitions and Notation}
%-------------------------------------------------------------------------------------------------------------
\frame{\frametitle{Some Definitions}
	\begin{itemize}
	\item<1->	\textbf{Maturity}: Expiry $T$ of contract.
	\item<2->	\textbf{Default Time}: $\tau$, modelled as a random variable with values in $[0,\infty]$.
	\item<3->	\textbf{Face Value}: (or Nominal Value) ($N\geq 0$) of a bond is the promised payment at maturity. We often simply use $N=1$.
	\item<4->	\textbf{Recovery-Rate}: In case of default a fraction $\delta$ (or $R$) of the nominal is recovered. Often we model it as random variable with values in $[0,1]$.
	\item<5-> \textbf{Loss given default}: The  (percentage loss) in case of
	default, i.e. $LGD=(1-\delta)=(1-R)$.
	\item<6-> \textbf{Exposure at default} The amount $EAD$ the creditor is owed at
	the time of default.
	\item<7-> \textbf{Probability of default} Probability $PD$ that the obligor
	defaults on its obligations.
	\end{itemize}	
}

\begin{frame}[fragile]
\frametitle{Expected Loss}
The central quantity for credit risk is the \emph{expected loss} $EL$:
\begin{align}
	\text{EL} = \text{PD} \times \text{EAD} \times \text{LGD}.
\end{align}
\end{frame}


\frame{ \frametitle{Credit Events I  }
According the International Swaps and Derivatives Association (ISDA)
\begin{itemize}
				\item<1-> \textbf{Bankruptcy} Insolvency, or a legally-recognised situation where a debtor is unable to pay off his debts in full.
        \item<2-> \textbf{Obligation Acceleration} covers the situation, other than a Failure to Pay, where the relevant obligation becomes due and payable as a result of a default by the reference entity before the time when such obligation would otherwise have been due and payable.
		\item<3-> \textbf{Obligation Default} covers the situation, other than a Failure to Pay, where the relevant obligation becomes capable of being declared due and payable as a result of a default by the reference entity before the time when such obligation would otherwise have been capable of being so declared.
\end{itemize}
}
\frame{ \frametitle{Credit Events II}
\begin{itemize}

		\item<1-> \textbf{Failure to Pay} is defined to be a failure of the reference entity to make, when and where due, any payments under one or more obligations. Grace periods for payment are taken into account.

		\item<2-> \textbf{Repudiation/Moratorium} deals with the situation where the reference entity or a governmental authority disaffirms, disclaims or otherwise challenges the validity of the relevant obligation.
\item<3-> \textbf{Restructuring} covers events as a result of which the terms, as agreed by the reference entity or governmental authority and the holders of the relevant obligation, governing the relevant obligation have become less favourable to the holders that they would otherwise have been.
	\end{itemize}
}

\frame{ \frametitle{Pricing a Zero-Coupon Corporate Bond}

The general valuation formula for a corporate bond is
$$
p^d (0,T) =  \EX \left[ e^{- r(T-t)} \IF_{\{\tau >T\}} + e^{-(\tau-t)} \delta \IF_{\{\tau \leq T\}}\right]\,.
$$
This is
\begin{eqnarray*}
p^d(0,T)	&=& e^{-rT} \prob(\tau > T ) +  \EX\left[e^{-r\tau} \delta \IF_{\{0\leq \tau \leq T\}}\right]\\
			&=& e^{-rT}\prob(\tau>T)+\delta\int_0^T e^{-rt}d\prob(\tau\leq t).
\end{eqnarray*}
To do calculations we need a model which allows to specify (compute) $\prob(\tau\leq t)$.
}

\frame{ \frametitle{Pricing a CDS}
Since the cash flow at coupon date $t_i$ for the protection
buyer is $s\IF_{\{\tau>t_i\}}$ and the payment for the protection
seller at time of default $\tau$ is $(1-\delta)\IF_{\{\tau\leq
T\}}$, we obtain
$$
\EX\left(e^{-r\tau}(1-\delta)\IF_{\{\tau\leq
T\}}\right) = \DSE\sum_{i=1}^n e^{-rt_i} s\EX\left(\IF_{\{\tau
>t_i\}}\right),
$$
and so
$$
s=\frac{\DSE \EX\left(e^{-r\tau}(1-\delta)\IF_{\{\tau\leq
T\}}\right)} {\DSE\sum_{i=1}^n e^{-rt_i} \EX\left(\IF_{\{\tau
>t_i\}}\right)}.
$$

\frame{\frametitle{Definitions and notation}

	\item<2-> The default boundary is $d$.

\subsection{Merton Model}

\frame{
\frametitle{Basic Merton Model}

Merton (1974)
assumes that a firm is financed by
equity and a single zero-coupon bond with notational amount (face
value) $F$ and maturity $T$.

The firm's value is given by
$$
dV(t) = (r-\gamma) V(t)  dt + \sigma V(t) dW(t)
$$
under an equivalent martingale (pricing) measure $\prob,$ with $r,
\sigma$ constant, $W$ Brownian motion and constant payout
(dividend) rate $\gamma,$ which may be negative (i.e. pay-in).

}
\frame{ \frametitle{Basic Merton Model}

Default is only possible at maturity. There are two possibilities:
$$
V_T \geq F, \quad \mbox{thus} \quad p^d(T,T)=F
$$
or
$$
V_T < F, \quad \mbox{thus} \quad p^d(T,T)=V_T.
$$
}
\frame{ \frametitle{Equity Owners}

For equity owners the payoff  is
$$
S_T=\max\{V_T-F,0\}
$$
thus stocks can be viewed as call options on the value of the firm
with ($\bar{V}_t = e^{-\gamma(T-t)}V_t$)
$$
S_t=C_E(\bar{V}_t,F)=e^{-\gamma(T-t)}V_t\Phi(d_1)-Fe^{-r(T-t)}\Phi(d_2)
$$
and
$$
d_1=\frac{\log(V_t/F)+(r-\gamma+\sigma^2/2)(T-t)}{\sigma\sqrt{T-t}}=d_2+\sigma\sqrt{T-t}.
$$
}
\frame{ \frametitle{Bond Owners}

For bond owners the payoff is
$$
p^d(T,T)=F-\max\{F-V_T,0\}.
$$
This can be viewed as the difference
of a risk-free payment and a put option on the value of the firm
with
$$
p^d(t,T)=Fe^{-r(T-t)}-P_E(\bar{V}_t,F),
$$
where
$$
P_E(\bar{V}_t,F)=-V_t e^{-\gamma(T-t)}
\Phi(-d_1)+Fe^{-r(T-t)}\Phi(-d_2).
$$
So
$$
p^d(t,T)= V_t e^{-\gamma(T-t)} \Phi(-d_1)+
Fe^{-r(T-t)}\Phi(d_2).
$$


}
\frame{ \frametitle{Calibration }

Typically, we do not observe the value of a firm. Thus in order to
calculate the parameters, i.e. $V_0$ and $\sigma$,  relevant for
the dynamics of the firm's value, we have to use the identities
\begin{eqnarray*}
S_0 &=& V_0e^{-\gamma T}\Phi(d_1)-Fe^{-rT}\Phi(d_2),\\*[12pt] \sigma_s &=&
\frac{\Phi(d_1)V_0\sigma}{S_0}
\end{eqnarray*}
to obtain risk-neutral default probabilities
$$
\prob(V_T<F).
$$
This approach has been used by Moody's KMV, named after
the founders Kealhofer, McQuown and Vasicek, in their portfolio model of credit-risky assets.

}
\frame{ \frametitle{Credit Spreads}

For spreads  $S(t,T)$ we use $p^d(t,T)=e^{-(r+S(t,T))(T-t)}$ and
find (assuming F=1)
$$
S(t,T)=\frac{1}{T-t}\log\left\{\frac{1}{l_t}e^{-\gamma(T-t)}\Phi(-d_1)+\Phi(d_2)\right\},
$$
with
$$
l_t=\frac{F e^{-r(T-t)}}{V_t} = \frac{ e^{-r(T-t)}}{V_t}
$$
the leverage ratio.

We find
$$
\lim_{t\rightarrow T} S(t,T)= \left\{\begin{array}{ll}
+\infty, &\mbox{ on}\; \{V_T<F\}\\*[6pt]
0, &\mbox{ on}\; \{V_T>F\}
\end{array} \right.
$$
}

\frame{\frametitle{Modelling Approach}
	   	\begin{itemize}
		\item<1-> Typically, we try to obtain the Portfolio Loss distribution from a portfolio model. The portfolio loss distribution is needed for portfolio derivatives (CDOs), and also risk management (VaR) questions.
		\item<2-> Portfolio models try to capture empirical observations (stylized facts)
		\begin{enumerate}
			\item	Fluctuation of the default probabilities with the economic cycle. Typically some macro-variables are used in the model to achieve this.
			\item Default cluster (contageous defaults), defaults of firms can trigger defaults of further firms.
		\end{enumerate}
		%\item< -> Portfolio models try to explain the dependence among several companies.
		\item<3-> Often individual default probabilities are assumed to be known and only the dependence structure is modelled.
\end{itemize}
}

\subsection{Vasicek Single Factor Model}
\frame{\frametitle{Vasicek Model}

\begin{itemize}
\item<1-> We now want to model a group of firms based on the
Merton model;
\item<2-> The different firm-value processes are
assumed to evolve according to correlated geometric
    Brownian motions, default is tested only at maturity.

\end{itemize}
}

\frame{\frametitle{Vasicek Model}

\begin{itemize}
\item<1-> The model for each individual firm-value process is
given by:
    $$dV_t^i=V_t^i(\gamma^i dt+\sigma^idW^i_t),\quad v_0>0.$$
\item<2-> Using It\^{o}'s formula we find:
    $$V_T^i=V_t^i\exp\left\{\left(\gamma^i-\frac{(\sigma^i)^2}{2}\right)(T-t)+\sigma^i\sqrt{T-t}X_{t}^i\right\},$$
    where $X_{t}^i=(W^i_T-W^i_t)/\sqrt{T-t}$ follows a standard normal distribution.
     \item<3-> The default threshold is $d^i$.
\end{itemize}
}

\frame{\frametitle{Market Factor}

\begin{itemize}
\item<1-> To incorporate correlation among different companies, we
partially explain $X_{t}^i$ by the common market factor $M_t$
    and idiosyncratic risk factors $\epsilon_t^i$.
\item<2-> So
    $$X_t^i=\rho M_t+\sqrt{1-\rho^2}\epsilon^i_t,\quad \rho\in[0,1],$$
    where $M_t, \epsilon_t^1,\dots,\epsilon_t^I$ are i.i.d. with $\mathcal{N}(0,1)$ distribution
\item<3-> So $\mbox{Cor}(X^i_t,X^k_t)=\rho$ for $i\neq k$.
\item<4-> Each $X_t^i$ is again distributed according to the
standard normal law.
\end{itemize}
}

\frame{\frametitle{Conditional Distribution}

\begin{itemize}
\item<1-> By conditioning on the common market factor default
probabilities of different firms are independent. \item<2-> We
denote those conditioned default probability by $p^i(M_t)$ and
obtain
    $$p^i(M_t)=\prob(\tau^i<t|M_t)=\Phi\left(\frac{k^i_t-\rho M_t}{\sqrt{1-\rho^2}}\right),$$
    where
    $$k^i_t=\frac{\log{\frac{d^i}{V_0^i}}-\left(\gamma^i-\frac{(\sigma^i)^2}{2}\right)t}{\sigma^i\sqrt{t}}.$$
\end{itemize}
}


\frame{\frametitle{Default Probabilities}

\begin{itemize}
\item<1-> We assume the term structure of individual default
probabilities as given and set $\Phi^{-1}(p^i_t)=k^i_t$. \item<2->
Furthermore, we assume all companies to have the same default
probabilities, the same portfolio weight
    and the same recovery rate. That is we model a homogeneous portfolio.
\item<3-> Therefore, we have
    $$p^i(M_t)=p(M_t)=\Phi\left(\frac{\Phi^{-1}(p_t)-\rho M_t}{\sqrt{1-\rho^2}}\right).$$
\end{itemize}
}

\frame{\frametitle{Loss Distribution}

\begin{itemize}
\item<1-> We now define $\Omega_t$ as the random variable that
describes the fraction of defaults in the portfolio
    up to time $t$.
\item<2-> The distribution of $\Omega_t$ depends on two
parameters: The individual default probabilities $p_t$
    and the correlation $\rho$ among different companies.
\item<3-> This distribution will be denoted by
    $$F_{p_t,\rho}(x)=\prob(\Omega_t\leq x).$$
\end{itemize}
}

\frame{\frametitle{Large Portfolio Approximation}

\begin{itemize}
\item<1-> We assume the number of companies within the portfolio
is very large (tends to infinity). \item<2-> By the law of large
numbers we can approximate $\prob(\Omega_t\leq x)$ by $p(M_t)$.
\item<3->  We show that
    $$F_{p_t,\rho}(x)\approx \Phi\left(\frac{\sqrt{1-\rho^2}\Phi^{-1}(x)-\Phi^{-1}(p_t)}{\rho}\right).$$
\item<3-> This approximation is continuous and strictly increasing
in $x$. As it further maps the unit interval
    onto itself it is a distribution function, too.
\end{itemize}
}

\frame{\frametitle{Large Portfolio Approximation -- Calculation}

\begin{itemize}
	\item<1-> We have
		\begin{eqnarray*}
		\prob(\Omega_t\leq x)	&=& \int_{-\infty}^{\infty}\prob(\Omega_t\leq x|M_t=y)d\prob(M_t\leq y)\\
							&=& \int_{-\infty}^{\infty}\IF_{p(y)}(x)d\prob(M_t\leq y) = \Phi(y^*),
		\end{eqnarray*}
		where $p(-y^*):=x$.
	\item<2-> So,
		$$\Phi\left(\frac{\Phi^{-1}(p_t)+\rho y^*}{\sqrt{1-\rho^2}}\right)=x\quad
		\Leftrightarrow\quad y^*=\frac{\sqrt{1-\rho^2}\Phi^{-1}(x)-\Phi^{-1}(p_t)}{\rho}.$$
	\item<3-> This shows
		$$F_{p_t,\rho}(x)\approx \Phi\left(\frac{\sqrt{1-\rho^2}\Phi^{-1}(x)-\Phi^{-1}(p_t)}{\rho}\right).$$
	\end{itemize}
}

\subsection{General Factor Models}

\frame{\frametitle{Gaussian Model}

\begin{itemize}
\item<1->
    $$X^i=\rho M+\sqrt{1-\rho^2}\epsilon^i,\quad \rho\in[0,1],$$
    where $M, \epsilon^1,\dots,\epsilon^I$ are i.i.d. with $\mathcal{N}(0,1)$ distribution
\item<2-> Therefore, we have
    $$p^i(M)=\Phi\left(\frac{\Phi^{-1}(F_i(t))-\rho M}{\sqrt{1-\rho^2}}\right).$$
\item<3-> No tail dependence.
\end{itemize}
}



\frame{\frametitle{Double-t model}

\begin{itemize}
\item<1-> Now
    $$X^i=\rho \left(\frac{\nu-2}{\nu}\right)^\frac{1}{2} M+
    \sqrt{1-\rho^2}\left(\frac{\bar{\nu}-2}{\bar{\nu}}\right)^\frac{1}{2}T^i$$
    and $M \sim t_\nu; T^i\sim t_{\bar{\nu}}$.
\item<2-> Joint distribution is obtained by simulation.
\end{itemize}
}
